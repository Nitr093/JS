GIT INIT
GIT CONFIG --GLOBAL USER.NAME "MILANI" / USER.EMAIL "INFERNAL@GMAIL.COM"
GIT CONFIG --GLOBAL COLOR.UI
GIT CONFIG --LOCAL / -- GLOBAL / --SYSTEM # local es a nivel loca. Global a todo y System a todos los usuarios.

GIT LOG #Se ve un historial de las modificaciones
GIT CONFIG LIST #Muestra la configuracion 
GIT CONFIG LIST --GLOBAL #Muestra la configuracion GLOBAL

GIT ../ #Vuelve para atras

mkdir #Para hacer directorio nuevo
rmdir #Para eliminar directorio
pwd #Para saber en que directorio estamos ubicados
ls -a #permite ver archivos ocultos (por si estamos en otro sistema)
